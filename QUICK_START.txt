📋 PREREQUISITES:
✓ cPanel hosting with PHP 7.4+ and MySQL
✓ Facebook Business Account
✓ Basic knowledge of editing text files

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

STEP 1: SETUP DATABASE (5 minutes)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. Open your cPanel
2. Go to "MySQL Databases"
3. Create database: "fb_messenger_bot"
4. Create user: "fb_bot" with strong password
5. Add user to database with ALL PRIVILEGES
6. Go to phpMyAdmin
7. Select your database
8. Click "Import"
9. Upload the "database.sql" file
10. Click "Import" button

✓ Database ready!

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

STEP 2: UPLOAD FILES (5 minutes)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. Open cPanel File Manager
2. Go to public_html
3. Create folder: "fb-bot"
4. Upload ALL FILES from the package into this folder
5. Include these files:
   - config.php
   - functions.php
   - webhook.php
   - index.php
   - admin/ (entire folder)
   - database.sql
   - documentation files
   - .htaccess
   - .gitignore

✓ Files uploaded!

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

STEP 3: CONFIGURE (5 minutes)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. Open File Manager → fb-bot → config.php
2. Find these lines and UPDATE THEM:

   Line 15-19 (Database):
   - DB_HOST: 'localhost' (keep as is)
   - DB_USER: 'fb_bot'
   - DB_PASS: (use the password you created)
   - DB_NAME: 'fb_messenger_bot'

   Line 24-28 (Facebook):
   - FB_PAGE_ID: Get from Facebook Developers
   - FB_PAGE_ACCESS_TOKEN: Get from Facebook Developers
   - FB_VERIFY_TOKEN: Create any random string (e.g., "verify123")

   Line 31 (Your Domain):
   - APP_URL: 'https://yourdomain.com'

3. Save the file

✓ Configuration complete!

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

STEP 4: TEST ADMIN PANEL (1 minute)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. Visit: https://yourdomain.com/fb-bot/admin/
2. Login with:
   Username: admin
   Password: admin123

If you see the dashboard → EXCELLENT! ✓

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

STEP 5: SETUP FACEBOOK WEBHOOK (10 minutes)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. Go to Facebook Developers: https://developers.facebook.com
2. Select your app
3. Go to Messenger → Settings
4. Scroll to "Webhooks"
5. Click "Add Callback URL"
6. Fill these fields:
   - Callback URL: https://yourdomain.com/fb-bot/webhook.php
   - Verify Token: (use the same as config.php FB_VERIFY_TOKEN)
7. Click "Verify and Save"

If status shows "✓ Verified" → SUCCESS! ✓

8. Find "Webhook Fields" section
9. Check these boxes:
   ☑ messages
   ☑ messaging_postbacks
   ☑ messaging_quick_replies

10. Select your Page and click "Subscribe"

✓ Webhook connected!

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

STEP 6: FIRST TEST (2 minutes)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. Go to your Facebook Page
2. Click "Send Message" button
3. Type: "hello"
4. You should get a welcome message back!

If you get a response → BOT IS WORKING! 🎉

5. Try typing "order" to test order flow
6. Check admin dashboard to see the order

🔐 IMPORTANT SECURITY STEPS
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. CHANGE ADMIN PASSWORD IMMEDIATELY!
   - Login to admin panel
   - Go to Settings
   - Click "Change Password"
   - Use strong password (12+ chars, mixed case, numbers)

2. BACKUP YOUR DATABASE
   - Set up weekly automated backups in cPanel

3. USE HTTPS ONLY
   - Get SSL certificate (free with cPanel)
   - Update config.php APP_URL to use https://

❓ COMMON ISSUES
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Issue: Admin login fails
→ Check DB_USER and DB_PASS in config.php match database

Issue: Webhook verification fails
→ Check FB_VERIFY_TOKEN in config.php matches Facebook
→ Check webhook URL is correct (must be HTTPS)

Issue: Bot doesn't respond
→ Check webhook is verified in Facebook Settings
→ Check "messages" event is subscribed
→ Check logs/ folder for errors

Issue: Database connection error
→ Verify MySQL is running in cPanel
→ Verify user has access to database
→ Test connection in phpMyAdmin
